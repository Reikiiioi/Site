/* ═══════════════════════════════════════════════

const SESSION_KEY = 'sc_session';

function saveSession(user) {
  try { localStorage.setItem(SESSION_KEY, JSON.stringify(user)); } catch(e){}
}
function loadSession() {
  try { return JSON.parse(localStorage.getItem(SESSION_KEY) || 'null'); } catch(e){ return null; }
}
function deleteSession() {
  try { localStorage.removeItem(SESSION_KEY); } catch(e){}
}
function clearAllSessions() {
  if(!confirm('Сбросить все сохранённые сессии?')) return;
  deleteSession();
  notify('🔑','Все сессии сброшены');
}

/* ═══════════════════════════════════════════════
   UV / THREE.JS HELPERS
═══════════════════════════════════════════════ */
function px2uv(px,py,pw,ph,tw,th){
  return [px/tw,py/th,(px+pw)/tw,py/th,px/tw,(py+ph)/th,(px+pw)/tw,(py+ph)/th];
}
function makeBox(w,h,d,faceUVs){
  const geo=new THREE.BoxGeometry(w,h,d);
  const uva=geo.attributes.uv;
  for(let f=0;f<6;f++){const fu=faceUVs[f],b=f*4;uva.setXY(b,fu[0],fu[1]);uva.setXY(b+1,fu[2],fu[3]);uva.setXY(b+2,fu[4],fu[5]);uva.setXY(b+3,fu[6],fu[7]);}
  uva.needsUpdate=true;return geo;
}
function mcFaceUVs(ox,oy,w,h,d,tw,th){
  return [
    px2uv(ox,oy+d,d,h,tw,th),
    px2uv(ox+d+w,oy+d,d,h,tw,th),
    px2uv(ox+d,oy,w,d,tw,th),
    px2uv(ox+d+w,oy,w,d,tw,th),
    px2uv(ox+d,oy+d,w,h,tw,th),
    px2uv(ox+d+w+d,oy+d,w,h,tw,th),
  ];
}

/* ═══════════════════════════════════════════════
   SKIN SCENE — 3D рендер скина
   ИСПРАВЛЕНИЕ: прозрачные пиксели = полностью прозрачны,
   без чёрной заливки.
   Используем premultipliedAlpha:false + alphaTest.
═══════════════════════════════════════════════ */
class SkinScene {
  constructor(canvas, opts) {
    this.canvas=canvas;
    this.opts=Object.assign({w:220,h:340,autoRotate:true,walk:true,small:false},opts);
    this.t=0;this.animId=null;this.isDragging=false;
    this.lastMX=0;this.lastMY=0;this.rotY=0.3;this.rotX=0.05;this._tex=null;
    this._initRenderer();this._initScene();this._buildModel(null,true);this._bindDrag();this._loop();
  }

  _initRenderer(){
    this.renderer=new THREE.WebGLRenderer({
      canvas:this.canvas,
      alpha:true,
      antialias:false,
      premultipliedAlpha:false   // ← ключевой флаг против чёрных краёв
    });
    this.renderer.setPixelRatio(Math.min(window.devicePixelRatio||1,2));
    this.renderer.setSize(this.opts.w,this.opts.h,false);
    this.renderer.setClearColor(0x000000,0);
    this.renderer.sortObjects=true;
  }

  _initScene(){
    this.scene=new THREE.Scene();
    this.camera=new THREE.PerspectiveCamera(40,this.opts.w/this.opts.h,0.1,200);
    this.camera.position.set(0,0,this.opts.small?32:26);
    this.scene.add(new THREE.AmbientLight(0xffffff,0.6));
    const sun=new THREE.DirectionalLight(0xffffff,0.85);sun.position.set(2,4,5);this.scene.add(sun);
    const fill=new THREE.DirectionalLight(0x8899cc,0.2);fill.position.set(-3,-1,-3);this.scene.add(fill);
    this.pivot=new THREE.Group();this.scene.add(this.pivot);
  }

  setSkin(dataURL){
    const img=new Image();img.crossOrigin='anonymous';
    img.onload=()=>{
      if(this._tex)this._tex.dispose();
      this._tex=new THREE.Texture(img);
      this._tex.flipY=false;
      this._tex.magFilter=THREE.NearestFilter;
      this._tex.minFilter=THREE.NearestFilter;
      this._tex.generateMipmaps=false;
      this._tex.needsUpdate=true;
      this._buildModel(this._tex,img.height>=64);
    };
    img.src=dataURL;
  }

  /* ─── Создание материалов ───
     alphaTest=0.5 — пиксели с альфой < 0.5 полностью отбрасываются.
     Это устраняет полупрозрачные/чёрные края.
  */
  _mkMat(tex,isOverlay){
    if(!tex){
      return new THREE.MeshLambertMaterial({color:0x888888,transparent:true,opacity:.5});
    }
    if(isOverlay){
      return new THREE.MeshLambertMaterial({
        map:tex,
        transparent:true,
        alphaTest:0.1,        // оверлей — мягкая отсечка
        depthWrite:false,
        side:THREE.FrontSide,
      });
    }
    return new THREE.MeshLambertMaterial({
      map:tex,
      transparent:false,      // основная геометрия непрозрачна
      alphaTest:0.5,           // отсекаем прозрачные пиксели (1-битная прозрачность)
      side:THREE.FrontSide,
    });
  }

  _buildModel(tex,isNew){
    while(this.pivot.children.length){
      const c=this.pivot.children[0];
      c.traverse(o=>{if(o.geometry)o.geometry.dispose();if(o.material&&o!==c){if(o.material.map)o.material.map.dispose();o.material.dispose();}});
      this.pivot.remove(c);
    }
    const TW=64,TH=isNew?64:32;
    const mSolid=this._mkMat(tex,false);
    const mOver =this._mkMat(tex,true);

    const S=this.opts.small?0.48:0.72;
    const HW=8*S,HH=8*S,HD=8*S,BW=8*S,BH=12*S,BD=4*S;
    const AW=4*S,AH=12*S,AD=4*S,LW=4*S,LH=12*S,LD=4*S;
    const totalH=HH+BH+LH,yBase=-totalH/2;
    const yBody=yBase+LH+BH/2,yHead=yBase+LH+BH+HH/2;

    // HEAD
    this.headMesh=new THREE.Mesh(makeBox(HW,HH,HD,mcFaceUVs(0,0,8,8,8,TW,TH)),mSolid);
    this.headMesh.position.set(0,yHead,0);
    this.headMesh.renderOrder=0;
    this.pivot.add(this.headMesh);
    if(tex){
      const hat=new THREE.Mesh(makeBox(HW*1.13,HH*1.13,HD*1.13,mcFaceUVs(32,0,8,8,8,TW,TH)),mOver);
      hat.position.copy(this.headMesh.position);hat.renderOrder=1;this.pivot.add(hat);
    }

    // BODY
    this.bodyMesh=new THREE.Mesh(makeBox(BW,BH,BD,mcFaceUVs(16,16,8,12,4,TW,TH)),mSolid);
    this.bodyMesh.position.set(0,yBody,0);this.bodyMesh.renderOrder=0;this.pivot.add(this.bodyMesh);
    if(tex&&isNew){
      const ov=new THREE.Mesh(makeBox(BW*1.08,BH*1.08,BD*1.08,mcFaceUVs(16,32,8,12,4,TW,TH)),mOver);
      ov.position.copy(this.bodyMesh.position);ov.renderOrder=1;this.pivot.add(ov);
    }

    // RIGHT ARM
    this.rightArm=new THREE.Group();this.rightArm.position.set(-(BW/2+AW/2),yBase+LH+BH,0);
    const raMesh=new THREE.Mesh(makeBox(AW,AH,AD,mcFaceUVs(40,16,4,12,4,TW,TH)),mSolid);
    raMesh.position.set(0,-AH/2,0);raMesh.renderOrder=0;this.rightArm.add(raMesh);this.pivot.add(this.rightArm);
    if(tex&&isNew){
      const ov=new THREE.Mesh(makeBox(AW*1.12,AH*1.12,AD*1.12,mcFaceUVs(40,32,4,12,4,TW,TH)),mOver);
      ov.position.copy(raMesh.position);ov.renderOrder=1;this.rightArm.add(ov);
    }

    // LEFT ARM
    this.leftArm=new THREE.Group();this.leftArm.position.set(BW/2+AW/2,yBase+LH+BH,0);
    const laUV=(tex&&isNew)?mcFaceUVs(32,48,4,12,4,TW,TH):mcFaceUVs(40,16,4,12,4,TW,TH);

    this.leftArm.add(laMesh);this.pivot.add(this.leftArm);
    if(tex&&isNew){
      const ov=new THREE.Mesh(makeBox(AW*1.12,AH*1.12,AD*1.12,mcFaceUVs(48,48,4,12,4,TW,TH)),mOver);
      ov.position.copy(laMesh.position);ov.renderOrder=1;this.leftArm.add(ov);
    }

    // RIGHT LEG
    this.rightLeg=new THREE.Group();this.rightLeg.position.set(-LW/2,yBase+LH,0);
    const rlMesh=new THREE.Mesh(makeBox(LW,LH,LD,mcFaceUVs(0,16,4,12,4,TW,TH)),mSolid);
    rlMesh.position.set(0,-LH/2,0);rlMesh.renderOrder=0;this.rightLeg.add(rlMesh);this.pivot.add(this.rightLeg);
    if(tex&&isNew){
      const ov=new THREE.Mesh(makeBox(LW*1.12,LH*1.12,LD*1.12,mcFaceUVs(0,32,4,12,4,TW,TH)),mOver);
      ov.position.copy(rlMesh.position);ov.renderOrder=1;this.rightLeg.add(ov);
    }

    // LEFT LEG
    this.leftLeg=new THREE.Group();this.leftLeg.position.set(LW/2,yBase+LH,0);
    const llUV=(tex&&isNew)?mcFaceUVs(16,48,4,12,4,TW,TH):mcFaceUVs(0,16,4,12,4,TW,TH);
    const llMesh=new THREE.Mesh(makeBox(LW,LH,LD,llUV),mSolid);
    llMesh.position.set(0,-LH/2,0);if(!isNew)llMesh.scale.x=-1;llMesh.renderOrder=0;
    this.leftLeg.add(llMesh);this.pivot.add(this.leftLeg);
    if(tex&&isNew){
      const ov=new THREE.Mesh(makeBox(LW*1.12,LH*1.12,LD*1.12,mcFaceUVs(0,48,4,12,4,TW,TH)),mOver);
      ov.position.copy(llMesh.position);ov.renderOrder=1;this.leftLeg.add(ov);
    }

    this.pivot.position.y=totalH/2-LH;
  }

  _bindDrag(){
    const c=this.canvas;
    const dn=(x,y)=>{this.isDragging=true;this.lastMX=x;this.lastMY=y;};
    const mv=(x,y)=>{if(!this.isDragging)return;this.rotY+=(x-this.lastMX)*.012;this.rotX+=(y-this.lastMY)*.006;this.rotX=Math.max(-.45,Math.min(.45,this.rotX));this.lastMX=x;this.lastMY=y;};
    const up=()=>{this.isDragging=false;};
    c.addEventListener('mousedown',e=>dn(e.clientX,e.clientY));
    window.addEventListener('mousemove',e=>mv(e.clientX,e.clientY));
    window.addEventListener('mouseup',up);
    c.addEventListener('touchstart',e=>{e.preventDefault();dn(e.touches[0].clientX,e.touches[0].clientY);},{passive:false});
    window.addEventListener('touchmove',e=>{if(e.touches[0])mv(e.touches[0].clientX,e.touches[0].clientY);},{passive:true});
    window.addEventListener('touchend',up);
  }

  _loop(){
    this.animId=requestAnimationFrame(()=>this._loop());
    this.t+=0.04;
    if(!this.isDragging&&this.opts.autoRotate) this.rotY+=0.01;
    this.pivot.rotation.y=this.rotY;
    this.pivot.rotation.x=this.rotX;
    if(this.opts.walk&&this.rightArm){
      const sw=Math.sin(this.t)*.42;
      this.rightArm.rotation.x=sw;this.leftArm.rotation.x=-sw;
      this.rightLeg.rotation.x=-sw*.75;this.leftLeg.rotation.x=sw*.75;
      if(this.headMesh)this.headMesh.rotation.y=Math.sin(this.t*.5)*.06;
    }
    this.renderer.render(this.scene,this.camera);
  }

  dispose(){
    if(this.animId)cancelAnimationFrame(this.animId);
    this.pivot.traverse(o=>{
      if(o.geometry)o.geometry.dispose();
      if(o.material){if(o.material.map)o.material.map.dispose();o.material.dispose();}
    });
    if(this._tex)this._tex.dispose();
    this.renderer.dispose();
  }
}

function updateHeroSwitcher(){
  const sw=document.getElementById('heroSwitcher');
  const counter=document.getElementById('hswCounter');
  const prevBtn=document.getElementById('hswPrev');
  const nextBtn=document.getElementById('hswNext');
  const nameEl=document.getElementById('heroSkinName');

  if(!skins.length){
    if(sw)sw.style.display='none';
    if(nameEl)nameEl.textContent='— нет скинов —';
    if(authEl)authEl.textContent='by kapustiano';
    return;
  }
  if(sw)sw.style.display='flex';

  // clamp index
  heroSkinIndex=Math.max(0,Math.min(heroSkinIndex,skins.length-1));
  const s=skins[heroSkinIndex];
  if(counter)counter.textContent=`${heroSkinIndex+1} / ${skins.length}`;
  if(prevBtn)prevBtn.disabled=(heroSkinIndex===0);
  if(nextBtn)nextBtn.disabled=(heroSkinIndex===skins.length-1);
  if(nameEl)nameEl.textContent=s.name;
  if(authEl)authEl.textContent=`by ${s.author}`;
}

function cardHTML(s,adminMode){
  const adm=(adminMode||isAdmin)?`<button class="btn btn-red" style="width:100%;font-size:.3rem;padding:7px;margin-top:8px;" onclick="delSkin(${s.id})">🗑️ Удалить</button>`:'';

function delSkin(id){
  if(!confirm('Удалить скин?'))return;
  if(cardScenes[id]){try{cardScenes[id].dispose();}catch(e){}delete cardScenes[id];}
  const idx=skins.findIndex(s=>s.id===id);
  skins=skins.filter(s=>s.id!==id);
  if(heroSkinIndex>=skins.length)heroSkinIndex=Math.max(0,skins.length-1);
  save();disposeAllCards();renderShop();renderAdminSkins();renderAdmin();updateStats();
  initHeroScene();
  notify('🗑️','Скин удалён');
}
